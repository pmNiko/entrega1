def initialize
  alfabeto = ('a'..'z').to_a.join + ('A'..'Z').to_a.join
  numeros = ('0'..'9').to_a.join
  @alfanumerico = alfabeto + numeros
  @cipher = @alfanumerico.chars.rotate(3).join
end
